Color Magic

const btn = document.getElementById('btn'); const text = document.getElementById('num'); const nameInput = document.getElementById('userName'); btn.onclick = function() { const name = nameInput.value; // Create a random color const randomColor = '#' + Math.floor(Math.random()*16777215).toString(16); if (name === "") { text.innerText = "Enter a name!"; text.style.color = "red"; } else { text.innerText = "Style for " + name; text.style.color = randomColor; btn.style.backgroundColor = randomColor; } };
