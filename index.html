// app.js import { fetchCryptoPrices } from './api.js'; // 1. APP STATE let cryptoData = []; // 2. THE UI RENDERER function renderDashboard() { const container = document.getElementById('dashboardContainer'); const avgDisplay = document.getElementById('averagePriceDisplay'); container.innerHTML = ""; // Clear loader let totalCost = 0; cryptoData.forEach(coin => { const price = parseFloat(coin.priceUsd); const change = parseFloat(coin.changePercent24Hr).toFixed(2); const isPositive = change >= 0; totalCost += price; // Aggregate running sum // Create card element const card = document.createElement('div'); card.className = "bg-zinc-900 border border-zinc-800 p-6 rounded-3xl flex justify-between items-center hover:border-zinc-700 transition-all shadow-xl shadow-black/40"; card.innerHTML = `
${coin.symbol}

${coin.name}

$${price.toFixed(2)}

${isPositive ? '▲' : '▼'} ${Math.abs(change)}%

`; container.appendChild(card); }); // Calculate and display the average price of top assets const averagePrice = cryptoData.length > 0 ? (totalCost / cryptoData.length) : 0; avgDisplay.innerText = `$${averagePrice.toFixed(2)}`; } // 3. THE REFRESH CONTROLLER async function handleRefresh() { const refreshBtn = document.getElementById('refreshBtn'); // UI Feedback: Disable button while fetching data refreshBtn.innerText = "UPDATING..."; refreshBtn.disabled = true; refreshBtn.classList.add('opacity-50', 'cursor-not-allowed'); // Fetch new data stream cryptoData = await fetchCryptoPrices(); renderDashboard(); // Reset button UI state refreshBtn.innerText = "REFRESH"; refreshBtn.disabled = false; refreshBtn.classList.remove('opacity-50', 'cursor-not-allowed'); } // 4. INITIALIZATION FUNCTION async function initApp() { const container = document.getElementById('dashboardContainer'); container.innerHTML = `

CONNECTING TO LIVE FEED...

`; document.getElementById('refreshBtn').addEventListener('click', handleRefresh); cryptoData = await fetchCryptoPrices(); renderDashboard(); } initApp();
